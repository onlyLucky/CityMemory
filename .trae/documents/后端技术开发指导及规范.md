## 三、命名规范

### 3.1 文件命名

| 类型 | 命名规范 | 示例 |
|------|----------|------|
| 控制器 | xxx.controller.ts | `user.controller.ts` |
| 服务 | xxx.service.ts | `user.service.ts` |
| 模型 | xxx.ts | `user.ts` |
| 中间件 | xxx.ts | `auth.ts` |
| 路由 | xxx.ts | `user.ts` |
| 工具 | xxx.ts | `crypto.ts` |
| 类型 | xxx.d.ts | `user.d.ts` |
| 测试 | xxx.test.ts | `user.test.ts` |

### 3.2 变量命名

| 类型 | 命名规范 | 示例 |
|------|----------|------|
| 普通变量 | camelCase | `userName`, `totalCount` |
| 常量 | UPPER_SNAKE_CASE | `MAX_PAGE_SIZE`, `JWT_SECRET` |
| 私有变量 | _camelCase | `_privateData` |
| 布尔值 | is/has/can + camelCase | `isValid`, `hasPermission` |
| 数组 | 复数形式 | `users`, `questions` |
| 对象 | 单数形式 | `user`, `question` |
| 函数 | 动词开头 | `getUserList`, `createQuestion` |
| 异步函数 | async前缀或动词 | `asyncGetUser`, `fetchUserData` |
| 类 | PascalCase | `UserService`, `QuestionController` |
| 接口 | I + PascalCase | `IUserInfo`, `IQuestionData` |
| 类型 | PascalCase | `UserInfo`, `QuestionData` |

### 3.3 数据库命名

| 类型 | 命名规范 | 示例 |
|------|----------|------|
| 表名 | snake_case（复数） | `users`, `questions` |
| 字段名 | snake_case | `user_name`, `created_at` |
| 主键 | id | `id` |
| 外键 | 表名_id | `user_id`, `question_id` |
| 索引 | idx_表名_字段 | `idx_users_status` |
| 唯一索引 | uk_表名_字段 | `uk_users_email` |

### 3.4 API命名

| 类型 | 命名规范 | 示例 |
|------|----------|------|
| 获取列表 | GET /resources | `GET /api/v1/admin/users` |
| 获取详情 | GET /resources/:id | `GET /api/v1/admin/users/:id` |
| 创建 | POST /resources | `POST /api/v1/admin/questions` |

## 四、编码规范

### 4.1 控制器规范

```typescript
import { Request, Response, NextFunction } from 'express'
import { UserService } from '@/services/admin/user.service'
import { success, error } from '@/utils/response'
import { logger } from '@/utils/logger'

export class UserController {
  private userService: UserService

  constructor() {
    this.userService = new UserService()
  }

  getList = async (req: Request, res: Response, next: NextFunction) => {
    try {
      const { page = 1, pageSize = 20, nickname, province } = req.query

      const result = await this.userService.getUserList({
        page: Number(page),
        pageSize: Number(pageSize),
        nickname: nickname as string,
        province: province as string
      })

      return success(res, result)
    } catch (err) {
      logger.error('获取用户列表失败', err)
      next(err)
    }
  }

  getDetail = async (req: Request, res: Response, next: NextFunction) => {
    try {
      const { id } = req.params

      const user = await this.userService.getUserById(id)

      if (!user) {
        return error(res, '用户不存在', 404)
      }

      return success(res, user)
    } catch (err) {
      logger.error('获取用户详情失败', err)
      next(err)
    }
  }
}

export default new UserController()
```

### 4.2 服务层规范

```typescript
import { UserModel } from '@/models/mysql/user'
import { RedisService } from '@/services/common/cache.service'
import { logger } from '@/utils/logger'

interface UserListParams {
  page: number
  pageSize: number
  nickname?: string
  province?: string
}

interface UserListResult {
  list: User[]
  total: number
  page: number
  pageSize: number
}

export class UserService {
  private userModel: typeof UserModel
  private redisService: RedisService

  constructor() {
    this.userModel = UserModel
    this.redisService = new RedisService()
  }

  async getUserList(params: UserListParams): Promise<UserListResult> {
    const { page, pageSize, nickname, province } = params
    const offset = (page - 1) * pageSize

    const where: Record<string, unknown> = {}
    if (nickname) {
      where.nickname = { like: `%${nickname}%` }
    }
    if (province) {
      where.province = province
    }

    const [list, total] = await Promise.all([
      this.userModel.findAll({
        where,
        offset,
        limit: pageSize,
        order: [['created_at', 'DESC']]
      }),
      this.userModel.count({ where })
    ])

    return {
      list,
      total,
      page,
      pageSize
    }
  }

  async getUserById(id: string): Promise<User | null> {
    const cacheKey = `user:${id}`
    
    const cached = await this.redisService.get(cacheKey)
    if (cached) {
      return JSON.parse(cached)
    }

    const user = await this.userModel.findByPk(id)
    
    if (user) {
      await this.redisService.setex(cacheKey, 3600, JSON.stringify(user))
    }

    return user
  }
}

export default new UserService()
```

### 4.4 中间件规范

#### 4.4.1 认证中间件

```typescript
import { Request, Response, NextFunction } from 'express'
import { verifyToken } from '@/utils/jwt'
import { error } from '@/utils/response'
import { AdminModel } from '@/models/mysql/admin'

export interface AuthRequest extends Request {
  user?: {
    id: string
    username: string
    roleId: string
  }
}

export const authMiddleware = async (
  req: AuthRequest,
  res: Response,
  next: NextFunction
) => {
  try {
    const authorization = req.headers.authorization

    if (!authorization || !authorization.startsWith('Bearer ')) {
      return error(res, '未登录', 401)
    }

    const token = authorization.replace('Bearer ', '')

    const decoded = verifyToken(token)
    if (!decoded) {
      return error(res, '登录已过期', 401)
    }

    const admin = await AdminModel.findByPk(decoded.id)
    if (!admin || admin.status !== 1) {
      return error(res, '账号已被禁用', 403)
    }

    req.user = {
      id: admin.id,
      username: admin.username,
      roleId: admin.roleId
    }

    next()
  } catch (err) {
    return error(res, '认证失败', 401)
  }
}
```

#### 4.4.2 权限中间件

```typescript
import { Request, Response, NextFunction } from 'express'
import { error } from '@/utils/response'
import { PermissionService } from '@/services/admin/permission.service'
import { AuthRequest } from './auth'

export const permissionMiddleware = (permission: string) => {
  return async (req: AuthRequest, res: Response, next: NextFunction) => {
    try {
      if (!req.user) {
        return error(res, '未登录', 401)
      }

      const permissionService = new PermissionService()
      const hasPermission = await permissionService.checkPermission(
        req.user.roleId,
        permission
      )

      if (!hasPermission) {
        return error(res, '无权限访问', 403)
      }

      next()
    } catch (err) {
      return error(res, '权限校验失败', 500)
    }
  }
}
```

#### 4.4.3 错误处理中间件

```typescript
import { Request, Response, NextFunction } from 'express'
import { logger } from '@/utils/response'
import { error } from '@/utils/response'

export interface AppError extends Error {
  statusCode?: number
  code?: string
}

export const errorHandler = (
  err: AppError,
  req: Request,
  res: Response,
  next: NextFunction
) => {
  const statusCode = err.statusCode || 500
  const message = err.message || '服务器内部错误'

  logger.error('Error:', {
    method: req.method,
    url: req.url,
    statusCode,
    message,
    stack: err.stack
  })

  return error(res, message, statusCode)
}

export const notFoundHandler = (req: Request, res: Response) => {
  return error(res, '接口不存在', 404)
}
```

  return async (req: Request, res: Response, next: NextFunction) => {
    await Promise.all(schemas.map((schema) => schema.run(req)))

    const errors = validationResult(req)

  }

### 7.2 JWT配置

```typescript
import jwt from 'jsonwebtoken'

const JWT_SECRET = process.env.JWT_SECRET || 'your-secret-key'
const JWT_EXPIRES_IN = '2h'
const JWT_REFRESH_EXPIRES_IN = '7d'

export interface TokenPayload {
  id: string
  username: string
  roleId: string
}

export const generateToken = (payload: TokenPayload): string => {
  return jwt.sign(payload, JWT_SECRET, { expiresIn: JWT_EXPIRES_IN })
}

export const generateRefreshToken = (payload: TokenPayload): string => {
  return jwt.sign(payload, JWT_SECRET, { expiresIn: JWT_REFRESH_EXPIRES_IN })
}

export const verifyToken = (token: string): TokenPayload | null => {
  try {
    return jwt.verify(token, JWT_SECRET) as TokenPayload
  } catch {
    return null
  }
}
```

### 7.4 XSS防护

```typescript
import xss from 'xss'

export const sanitizeInput = (input: string): string => {
  return xss(input, {
    whiteList: {},
    stripIgnoreTag: true,
    stripIgnoreTagBody: ['script']
  })
}

export const sanitizeObject = (obj: Record<string, any>): Record<string, any> => {
  const result: Record<string, any> = {}
  for (const key in obj) {
    if (typeof obj[key] === 'string') {
      result[key] = sanitizeInput(obj[key])
    } else if (typeof obj[key] === 'object' && obj[key] !== null) {
      result[key] = sanitizeObject(obj[key])
    } else {
      result[key] = obj[key]
    }
  }
  return result
}
```
