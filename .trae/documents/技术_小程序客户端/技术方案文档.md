## 二、技术选型

### 2.1 整体技术栈

| 层级 | 技术选型 | 版本要求 | 选型理由 |
|------|----------|----------|----------|
| 小程序框架 | 微信原生 | 基础库 2.20.0+ | 性能最优，API支持完整 |
| 状态管理 | MobX | 6.x | 轻量级，响应式，适合小程序 |
| UI组件库 | 自研组件 | - | 定制化需求，保持极简风格 |
| 后端语言 | Node.js | 18.x LTS | 开发效率高，生态丰富 |
| Web框架 | Koa | 2.x | 轻量灵活，中间件机制完善 |
| 关系数据库 | MySQL | 8.0 | 成熟稳定，事务支持完善 |
| 缓存数据库 | Redis | 7.0 | 高性能，支持排行榜、会话 |
| 文档数据库 | MongoDB | 6.0 | 题库数据灵活，Schema自由 |
| 对象存储 | 腾讯云COS | - | 微信生态友好，CDN加速 |
| 容器化 | Docker | 24.x | 标准化部署，便于扩展 |

### 2.2 小程序技术选型详解

#### 2.2.1 为什么选择原生开发

| 对比项 | 原生开发 | Uni-app | Taro |

| 性能 | ⭐⭐⭐⭐⭐ | ⭐⭐⭐⭐ | ⭐⭐⭐⭐ |
| API支持 | ⭐⭐⭐⭐⭐ | ⭐⭐⭐⭐ | ⭐⭐⭐⭐ |
| 包体积 | 最小 | 较大 | 较大 |
| 调试体验 | 最佳 | 良好 | 良好 |
| 学习成本 | 中等 | 较低 | 较低 |
| 跨平台 | 否 | 是 | 是 |

**结论**：城迹仅面向微信小程序，无需跨平台，原生开发性能最优。

### 4.2 门票系统模块

#### 4.2.1 门票恢复机制

```javascript
class TicketManager {
  constructor() {
    this.ADVENTURE_MAX = 30
    this.ADVENTURE_RECOVER_INTERVAL = 10 * 60 * 1000
  }

  async getAdventureTickets(userId) {
    const ticket = await redis.hgetall(`ticket:${userId}`)
    const now = Date.now()
    
    if (ticket.current < this.ADVENTURE_MAX) {
      const lastRecover = parseInt(ticket.last_recover_time) || now
      const elapsed = now - lastRecover
      const recoverCount = Math.floor(elapsed / this.ADVENTURE_RECOVER_INTERVAL)
      
      if (recoverCount > 0) {
        const newCount = Math.min(
          ticket.current + recoverCount,
          this.ADVENTURE_MAX
        )
        const newLastRecover = lastRecover + recoverCount * this.ADVENTURE_RECOVER_INTERVAL
        
        await redis.hmset(`ticket:${userId}`, {
          current: newCount,
          last_recover_time: newLastRecover
        })
        
        return { current: newCount, max: this.ADVENTURE_MAX }
      }
    }
    
    return { current: ticket.current, max: this.ADVENTURE_MAX }
  }

  async consumeTicket(userId) {
    const tickets = await this.getAdventureTickets(userId)
    if (tickets.current <= 0) {
      throw new Error('门票不足')
    }
    await redis.hincrby(`ticket:${userId}`, 'current', -1)
    return true
  }
}
```

#### 4.2.2 随机门票重置机制

```javascript
class RandomTicketManager {
  constructor() {
    this.RANDOM_MAX = 3
  }

  async getRandomTickets(userId) {
    const today = dayjs().format('YYYY-MM-DD')
    const key = `random_ticket:${userId}:${today}`
    
    let current = await redis.get(key)
    if (current === null) {
      await redis.set(key, this.RANDOM_MAX, 'EX', 86400)
      current = this.RANDOM_MAX
    }
    
    return { current: parseInt(current), max: this.RANDOM_MAX }
  }

  async consumeRandomTicket(userId) {
    const today = dayjs().format('YYYY-MM-DD')
    const key = `random_ticket:${userId}:${today}`
    
    const current = await redis.decr(key)
    if (current < 0) {
      await redis.incr(key)
      throw new Error('今日次数已用完')
    }
    return true
  }
}
```

### 4.3 星级计算模块

#### 4.3.1 星级计算算法

```javascript
class StarCalculator {
  static calculateStars(correctCount, totalCount, avgTime) {
    const accuracy = correctCount / totalCount
    const baseStars = this.getBaseStars(accuracy)
    const timeBonus = this.getTimeBonus(avgTime)
    const finalStars = Math.min(baseStars + timeBonus, 6)
    
    return Math.round(finalStars * 2) / 2
  }

  static getBaseStars(accuracy) {
    if (accuracy >= 1.0) return 5
    if (accuracy >= 0.9) return 4.5
    if (accuracy >= 0.8) return 4
    if (accuracy >= 0.7) return 3.5
    if (accuracy >= 0.6) return 3
    if (accuracy >= 0.5) return 2.5
    if (accuracy >= 0.4) return 2
    if (accuracy >= 0.3) return 1.5
    if (accuracy >= 0.2) return 1
    return 0.5
  }

  static getTimeBonus(avgTime) {
    if (avgTime <= 5) return 1
    if (avgTime <= 10) return 0.5
    return 0
  }

  static calculateStarDiff(oldStars, newStars) {
    if (newStars > oldStars) {
      return newStars - oldStars
    }
    return 0
  }
}
```

### 4.4 排行榜模块

#### 4.4.1 Redis排行榜设计

```javascript
class RankService {
  constructor() {
    this.STAR_RANK_KEY = 'rank:stars'
    this.LEVEL_RANK_KEY = 'rank:level'
  }

  async updateStarRank(userId, totalStars, levelCount) {
    const score = totalStars * 10000 + levelCount
    await redis.zadd(this.STAR_RANK_KEY, score, userId)
  }

  async getStarRank(userId) {
    const rank = await redis.zrevrank(this.STAR_RANK_KEY, userId)
    return rank + 1
  }

  async getStarRankList(start = 0, stop = 99) {
    const list = await redis.zrevrange(
      this.STAR_RANK_KEY, 
      start, 
      stop, 
      'WITHSCORES'
    )
    return this.parseRankList(list)
  }

  parseRankList(list) {
    const result = []
    for (let i = 0; i < list.length; i += 2) {
      const userId = list[i]
      const score = parseFloat(list[i + 1])
      result.push({
        userId,
        totalStars: Math.floor(score / 10000),
        levelCount: score % 10000,
        rank: i / 2 + 1
      })
    }
    return result
  }
}
```

### 4.5 本地题目生成模块

#### 4.5.1 题目生成器

```javascript
class QuestionGenerator {
  constructor() {
    this.cityProvinceData = null
    this.cityCountryData = null
    this.ancientNameData = null
    this.flagData = null
  }

  async init() {
    this.cityProvinceData = await this.loadData('city_province')
    this.cityCountryData = await this.loadData('city_country')
    this.ancientNameData = await this.loadData('ancient_name')
    this.flagData = await this.loadData('flag')
  }

  generateQuestion(type, region, difficulty) {
    switch (type) {
      case 1:
        return this.generateCityProvince(region, difficulty)
      case 2:
        return this.generateCityCountry(region, difficulty)
      case 3:
        return this.generateAncientName(region, difficulty)
      case 4:
        return this.generateFlagCountry(region, difficulty)
      default:
        throw new Error('Unknown question type')
    }
  }

  generateCityProvince(region, difficulty) {
    const cities = this.cityProvinceData.filter(c => 
      c.province === region || region === 'all'
    )
    const target = this.randomPick(cities)
    const distractors = this.generateProvinceDistractors(target.province, 3)
    const options = this.shuffleOptions([
      { key: 'A', value: target.province, correct: true },
      ...distractors.map((d, i) => ({ 
        key: ['B', 'C', 'D'][i], 
        value: d, 
        correct: false 
      }))
    ])

    return {
      question_id: this.generateId(),
      question_type: 1,
      question_content: `${target.city}属于哪个省份？`,
      options,
      correct_answer: options.find(o => o.correct).key,
      difficulty,
      region: target.province
    }
  }

  generateProvinceDistractors(correctProvince, count) {
    const provinces = this.cityProvinceData
      .map(c => c.province)
      .filter(p => p !== correctProvince)
    const unique = [...new Set(provinces)]
    return this.randomPickMultiple(unique, count)
  }

  shuffleOptions(options) {
    const shuffled = [...options]
    for (let i = shuffled.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1));
      [shuffled[i], shuffled[j]] = [shuffled[j], shuffled[i]]
    }
    return shuffled.map((o, i) => ({
      ...o,
      key: ['A', 'B', 'C', 'D'][i]
    }))
  }
}
```
