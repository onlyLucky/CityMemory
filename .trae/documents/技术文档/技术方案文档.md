# 城迹项目技术方案文档

## 一、文档概述

### 1.1 文档目的
本文档定义"城迹"项目的技术实现方案，包括小程序端、后台管理系统和服务端的详细技术实现，为开发团队提供明确的技术指导。

### 1.2 项目目标
- 构建一款地理知识问答类微信小程序
- 提供完整的运营后台管理系统
- 实现统一的服务端API
- 支持高并发、高可用的系统架构

### 1.3 技术选型总结

| 模块 | 技术栈 | 选型理由 |
|------|--------|----------|
| 小程序端 | UniApp + Vue 3 + Pinia | 跨平台开发，一次开发多端运行 |
| 后台管理 | Vue 3 + Element Plus | 企业级UI组件库，功能完善 |
| 服务端 | Node.js + Koa + TypeScript | 高性能，类型安全，生态丰富 |

// 使用Pinia缓存
const useUserStore = defineStore('user', () => {
  const userInfo = ref<UserInfo | null>(null)
  const cacheTime = ref(0)

  const getUserInfo = async (forceRefresh = false) => {
    const now = Date.now()
    const cacheDuration = 5 * 60 * 1000 // 5分钟

    if (!forceRefresh && userInfo.value && (now - cacheTime.value) < cacheDuration) {
      return userInfo.value
    }

    const info = await userApi.getUserInfo()
    userInfo.value = info
    cacheTime.value = now
    return info
  }

  return {
    userInfo,
    getUserInfo
  }
})

// 开始关卡
async startLevel(userId: string, levelId: string): Promise<IStartLevelResult> {
  const level = await this.levelModel.findById(levelId)
  if (!level) {
    throw new Error('关卡不存在')
  }

  const progress = await this.userProgressModel.findByUserId(userId)
  if (!progress.unlockedLevels.includes(levelId)) {
    throw new Error('关卡未解锁')
  }

  const questions = await this.generateQuestions(level)

  return {
    levelId,
    questions
  }
}

// 提交答案
async submitAnswer(userId: string, levelId: string, answer: IAnswer): Promise<IAnswerResult> {
  const question = await this.questionModel.findById(answer.questionId)
  if (!question) {
    throw new Error('题目不存在')
  }

  const isCorrect = question.correctAnswer === answer.answer
  const progress = await this.userProgressModel.findByUserId(userId)

  if (!isCorrect) {
    progress.blood--
  }

  const isLevelCompleted = progress.blood <= 0 || progress.currentQuestionIndex >= progress.totalQuestions

  return {
    isCorrect,
    correctAnswer: question.correctAnswer,
    blood: progress.blood,
    isLevelCompleted
  }
}

// 完成关卡
async completeLevel(userId: string, levelId: string, result: ICompleteResult): Promise<ICompleteLevelResult> {
  const progress = await this.userProgressModel.findByUserId(userId)
  const level = await this.levelModel.findById(levelId)

  const stars = this.calculateStars(result.blood, result.correctCount, result.totalCount)

  progress.levelStars[levelId] = Math.max(progress.levelStars[levelId] || 0, stars)
  progress.completedLevels.push(levelId)

  const nextLevel = await this.levelModel.findNextLevel(level.province, level.levelOrder)
  if (nextLevel && !progress.unlockedLevels.includes(nextLevel.id)) {
    progress.unlockedLevels.push(nextLevel.id)
  }

  await this.userProgressModel.update(userId, progress)

  const user = await this.userModel.findById(userId)
  user.totalStars += stars
  user.levelCount = progress.completedLevels.length
  await this.userModel.update(userId, user)

  return {
    levelId,
    stars,
    totalStars: user.totalStars,
    unlockedLevel: nextLevel?.id
  }
}
```

// 题目生成
async generateQuestions(level: Level): Promise<Question[]> {
  const usedIds = new Set<string>()
  const questions: Question[] = []

  for (let i = 0; i < level.questionCount; i++) {
    const question = await this.generateSingleQuestion(level.difficulty, level.province, usedIds)
    if (question) {
      questions.push(question)
    }
  }

  return questions
}

async generateSingleQuestion(difficulty: number, region: string, usedIds: Set<string>): Promise<Question | null> {
  const candidates = await this.questionModel.find({
    difficulty,
    region,
    status: 1,
    id: { $nin: Array.from(usedIds) }
  }).limit(100)

  if (candidates.length === 0) {
    return null
  }

  const question = candidates[Math.floor(Math.random() * candidates.length)]
  usedIds.add(question.id)

  return question
}
```

async getTicketStatus(userId: string): Promise<ITicketStatus> {
  const cacheKey = `ticket:${userId}`
  let tickets = await redisUtil.get(cacheKey)

  if (!tickets) {
    tickets = await this.initTickets(userId)
    await redisUtil.set(cacheKey, JSON.stringify(tickets), 86400)
  }

  return JSON.parse(tickets)
}

// 使用门票
async useTicket(userId: string, type: 'adventure' | 'random'): Promise<void> {
  const tickets = await this.getTicketStatus(userId)
  const ticketType = type === 'adventure' ? 'adventureTickets' : 'randomTickets'

  if (tickets[ticketType].current <= 0) {
    throw new Error('门票不足')
  }

  tickets[ticketType].current--

  const cacheKey = `ticket:${userId}`
  await redisUtil.set(cacheKey, JSON.stringify(tickets), 86400)
}

// 恢复门票
async recoverTickets(userId: string): Promise<void> {
  const tickets = await this.getTicketStatus(userId)

  if (tickets.adventureTickets.current < tickets.adventureTickets.max) {
    tickets.adventureTickets.current++
    tickets.adventureTickets.nextRecoverIn = 1800 // 30分钟

// 获取排行榜
async getRankList(type: string, province?: string, page: number = 1, pageSize: number = 20): Promise<IRankList> {
  const key = `rank:${type}:${province || 'all'}`
  const start = (page - 1) * pageSize
  const end = start + pageSize - 1

  const result = await redisUtil.zrevrange(key, start, end, 'WITHSCORES')
  const total = await redisUtil.zcard(key)

  const list = []
  for (let i = 0; i < result.length; i += 2) {
    const userId = result[i]
    const score = result[i + 1]
    const user = await this.userModel.findById(userId)

    list.push({
      rank: start + i / 2 + 1,
      userId,
      nickname: user.nickname,
      avatar: user.avatar,
      province: user.province,
      value: Number(score)
    })
  }

  return {
    list,
    total,
    page,
    pageSize
  }
}

// 获取用户排名
async getUserRank(userId: string, type: string, province?: string): Promise<number> {
  const key = `rank:${type}:${province || 'all'}`
  const rank = await redisUtil.zrevrank(key, userId)
  return rank !== null ? rank + 1 : -1
}

```typescript
// 一级缓存：内存缓存
const memoryCache = new Map<string, any>()

// 二级缓存：Redis缓存
async getFromCache(key: string): Promise<any> {
  // 先查内存缓存
  if (memoryCache.has(key)) {
    return memoryCache.get(key)
  }

  // 再查Redis缓存
  const cached = await redisUtil.get(key)
  if (cached) {
    const data = JSON.parse(cached)
    memoryCache.set(key, data)
    return data
  }

  return null
}

async setToCache(key: string, data: any, ttl: number): Promise<void> {
  // 设置内存缓存
  memoryCache.set(key, data)

  // 设置Redis缓存
  await redisUtil.set(key, JSON.stringify(data), ttl)
}
```
